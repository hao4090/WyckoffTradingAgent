import sqlite3
from pathlib import Path

db_path = Path.home() / '.wyckoff' / 'wyckoff.db'
conn = sqlite3.connect(str(db_path))
cursor = conn.cursor()

# 1. 查看 recommendation_tracking 表结构
print("=== TABLE SCHEMA ===")
cursor.execute("PRAGMA table_info(recommendation_tracking)")
for row in cursor.fetchall():
    print(row)

# 2. 查看最近几天的 recommend_date 分布
print("\n=== RECOMMEND_DATE DISTRIBUTION ===")
cursor.execute("SELECT recommend_date, COUNT(*) FROM recommendation_tracking GROUP BY recommend_date ORDER BY recommend_date DESC LIMIT 10")
for row in cursor.fetchall():
    print(row)

conn.close()
